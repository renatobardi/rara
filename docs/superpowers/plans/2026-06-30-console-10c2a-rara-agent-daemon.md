# CONSOLE-#10c2a: rara-agent Daemon (Claude Code CLI executor) + Task UI

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a Mac launchd daemon (`rara-agent`) that polls `agent_tasks`, spawns Claude Code CLI for each claimed task, and writes back status/result — plus a minimal task-creation UI in the agents page.

**Architecture:** rara-agent is a standalone Go module (like rara-transcribe) with direct pgxpool connection to Neon for the task queue (FOR UPDATE SKIP LOCKED claim), HTTP to rara-core for agent/skill metadata and distillation content. It exposes an `Executor` seam — v1 is `CLIExecutor` (spawns `claude -p`) — tested with a `FakeExecutor`. rara-core gains `UpdateAgentTask` + extended `AgentTaskRow` so the console can display results. The agents page grows a task panel (create + live list + result view).

**Tech Stack:** Go 1.26, pgx/v5, `os/exec` (claude CLI invocation), Svelte 5 runes, launchd plist.

## Global Constraints

- TDD mandatory: write failing test before implementation, `make test` must pass before commit.
- No real I/O in unit tests: use `FakeExecutor` / `MockStore` at the executor/store seam.
- `make lint` (go vet + staticcheck) must pass before every commit.
- New rara-agent module: `go.mod` with `module rara-agent`, isolated from sibling modules (no `go.work`).
- Migrations idempotent (`IF NOT EXISTS`, `IF EXISTS`, `ON CONFLICT`) and safe to re-run.
- Config from environment only — required vars call `log.Fatalf`, never panic.
- Executor='cli' is the only implemented variant; `executor='gateway'` is reserved for 10c2b.
- Skill `trusted=false` → write SKILL.md content only, never script files. `trusted=true` → also write `skill_files`.
- rara-core surface.go routes: use pattern `METHOD /path` with `h.handlerFunc`.
- No new HTTP routes in rara-core for claim/update — daemon goes direct to Neon.
- BFF (rara-console) is pass-through JSON proxy; no business logic in BFF.

```go
		VALUES ($1, $2, $3, $4, $5, $6, $7)

			description  = EXCLUDED.description,
```

```go
func (c *Core) UpdateAgentTask(ctx context.Context, id int, status, sessionID, workDir string, result json.RawMessage, errMsg string) error {
	if id <= 0 {
		return badInput("task id must be positive, got %d", id)
	}

	}
```

```go
func (d *pgxDatabase) UpdateAgentTask(ctx context.Context, id int, status, sessionID, workDir string, result json.RawMessage, errMsg string) error {
	const q = `
		UPDATE agent_tasks SET
			status      = $2,
			session_id  = NULLIF($3, ''),
			work_dir    = NULLIF($4, ''),
			result      = CASE WHEN $5::text IS NOT NULL THEN $5::jsonb ELSE result END,
			error       = NULLIF($6, ''),
			completed_at = CASE WHEN $2 IN ('done','failed','cancelled')
			                    THEN CURRENT_TIMESTAMP
			                    ELSE completed_at END
		WHERE id = $1`
	var resultStr *string
	if len(result) > 0 {
		s := string(result)
		resultStr = &s
	}
	if _, err := d.conn.Exec(ctx, q, id, status, sessionID, workDir, resultStr, errMsg); err != nil {
		return fmt.Errorf("update agent task: %w", err)
	}
	return nil
}
```

```go
// runOnce claims one task and runs it. Returns (true, nil) if a task was run,
// (false, nil) if the queue was empty, (false, err) on a fatal store error.
func runOnce(ctx context.Context, cfg config) (bool, error) {
	task, err := cfg.store.ClaimTask(ctx)
	if err != nil {
		return false, fmt.Errorf("claim: %w", err)
	}
	if task == nil {
		return false, nil // queue empty
	}

	agent, err := cfg.store.FetchAgent(ctx, task.AgentID)
	if err != nil {
		// Log but don't crash — mark task failed and move on.
		log.Printf("agent %d: fetch: %v", task.AgentID, err)
		_ = cfg.store.UpdateTask(ctx, task.ID, "failed", "", "", nil, err.Error())
		return true, nil
	}

	skills, err := cfg.store.FetchSkills(ctx, agent.SkillIDs)
	if err != nil {
		log.Printf("task %d: fetch skills: %v", task.ID, err)
		// Non-fatal: proceed with no skills.
		skills = nil
	}

	tc := TaskCtx{
		TaskID:      task.ID,
		AgentID:     task.AgentID,
		Instruction: task.Instruction,
		ContextRefs: task.ContextRefs,
		Agent:       agent,
		Skills:      skills,
	}

	// Mark running before invoking the executor.
	_ = cfg.store.UpdateTask(ctx, task.ID, "running", "", "", nil, "")

	result, execErr := cfg.exec.Run(ctx, tc)
	if execErr != nil {
		log.Printf("task %d: executor: %v", task.ID, execErr)
		_ = cfg.store.UpdateTask(ctx, task.ID, "failed", result.SessionID, result.WorkDir, nil, execErr.Error())
		return true, nil
	}

	if err := cfg.store.UpdateTask(ctx, task.ID, "done", result.SessionID, result.WorkDir, result.Output, ""); err != nil {
		log.Printf("task %d: mark done: %v", task.ID, err)
	}
	return true, nil
}

	if err != nil || interval <= 0 {
		log.Fatalf("AGENT_POLL_INTERVAL_S must be a positive integer, got %q", pollSec)
	}

	store, err := newPgxStore(context.Background(), dbURL, coreURL, coreTok)
	if err != nil {
		log.Fatalf("store: %v", err)
```

```go
package main

import (
	"fmt"
	"os"
	"path/filepath"
	"strings"
)

// BuildWorkdir creates a temporary directory under baseDir and populates it for
// Claude Code CLI execution: CLAUDE.md (agent instructions), per-skill SKILL.md,
// and supporting files — but only for trusted skills.
func BuildWorkdir(tc TaskCtx, baseDir string) (string, error) {
	dir, err := os.MkdirTemp(baseDir, fmt.Sprintf("task-%d-*", tc.TaskID))
	if err != nil {
		return "", fmt.Errorf("mktemp: %w", err)
	}

	// CLAUDE.md: agent system prompt
	claudeMD := fmt.Sprintf("# Agent: %s\n\n%s\n", tc.Agent.Name, tc.Agent.Instructions)
	if err := writeFile(dir, "CLAUDE.md", claudeMD); err != nil {
		return dir, err
	}

	// Per-skill SKILL.md: concatenate all skills (most agents have one).
	// Supporting files (scripts, configs) are written only when skill.Trusted=true.
	var skillContents strings.Builder
	for _, skill := range tc.Skills {
		fmt.Fprintf(&skillContents, "# %s\n\n%s\n\n", skill.Name, skill.Content)
		if skill.Trusted {
			for _, f := range skill.Files {
				if err := writeFile(dir, f.Path, f.Content); err != nil {
					return dir, fmt.Errorf("skill %d file %q: %w", skill.ID, f.Path, err)
				}
			}
		}
	}
	if skillContents.Len() > 0 {
		if err := writeFile(dir, "SKILL.md", skillContents.String()); err != nil {
			return dir, err
		}
	}

	return dir, nil
}

// writeFile writes content to name inside dir, creating parent directories as needed.
// Paths in name are cleaned and must stay under dir (no ../ traversal).
func writeFile(dir, name, content string) error {
	// Security: reject path traversal — paths in skill_files come from DB content.
	clean := filepath.Clean(name)
	if filepath.IsAbs(clean) || strings.HasPrefix(clean, "..") {
		return fmt.Errorf("invalid file path %q", name)
	}
	full := filepath.Join(dir, clean)
	if err := os.MkdirAll(filepath.Dir(full), 0o700); err != nil {
		return fmt.Errorf("mkdir %q: %w", filepath.Dir(full), err)
	}
	return os.WriteFile(full, []byte(content), 0o600)
}
```

```go
import (
	"context"
	"encoding/json"
	"errors"
	"fmt"
	"io"
	"net/http"
	"time"

	"github.com/jackc/pgx/v5"
	"github.com/jackc/pgx/v5/pgxpool"
)

type pgxStore struct {
	pool      *pgxpool.Pool
	coreURL   string
	coreToken string
	client    *http.Client
}

func newPgxStore(ctx context.Context, dbURL, coreURL, coreToken string) (*pgxStore, error) {
	cfg, err := pgxpool.ParseConfig(dbURL)
	if err != nil {
		return nil, fmt.Errorf("parse db url: %w", err)
	}
	// SimpleProtocol avoids prepared-statement caching issues with PgBouncer.
	cfg.ConnConfig.DefaultQueryExecMode = pgx.QueryExecModeSimpleProtocol
	pool, err := pgxpool.NewWithConfig(ctx, cfg)
	if err != nil {
		return nil, fmt.Errorf("connect db: %w", err)
	}
	return &pgxStore{
		pool:      pool,
		coreURL:   coreURL,
		coreToken: coreToken,
		client:    &http.Client{Timeout: 10 * time.Second},
	}, nil
}

// ClaimTask atomically pulls the next queued task for executor='cli', moving it queued→dispatched.
// Returns nil when the queue is empty.
func (s *pgxStore) ClaimTask(ctx context.Context) (*Task, error) {
	tx, err := s.pool.Begin(ctx)
	if err != nil {
		return nil, fmt.Errorf("claim: begin: %w", err)
	}
	defer tx.Rollback(ctx) //nolint:errcheck

	const sel = `
		SELECT t.id, t.agent_id, t.instruction, t.context_refs
		FROM agent_tasks t
		JOIN agents a ON a.id = t.agent_id
		WHERE t.status = 'queued'
		  AND a.deleted_at IS NULL
		  AND a.executor = 'cli'
		ORDER BY t.priority DESC, t.created_at, t.id
		FOR UPDATE OF t SKIP LOCKED
		LIMIT 1`
	var t Task
	err = tx.QueryRow(ctx, sel).Scan(&t.ID, &t.AgentID, &t.Instruction, &t.ContextRefs)
	if errors.Is(err, pgx.ErrNoRows) {
		return nil, nil // empty queue
	}
	if err != nil {
		return nil, fmt.Errorf("claim: select: %w", err)
	}
	if _, err := tx.Exec(ctx,
		`UPDATE agent_tasks SET status='dispatched', dispatched_at=NOW() WHERE id=$1`, t.ID); err != nil {
		return nil, fmt.Errorf("claim: update: %w", err)
	}
	return &t, tx.Commit(ctx)
}

// UpdateTask writes progress for a claimed task.
func (s *pgxStore) UpdateTask(ctx context.Context, id int, status, sessionID, workDir string, result json.RawMessage, errMsg string) error {
	var resultStr *string
	if len(result) > 0 {
		rs := string(result)
		resultStr = &rs
	}
	const q = `
		UPDATE agent_tasks SET
			status      = $2,
			session_id  = NULLIF($3, ''),
			work_dir    = NULLIF($4, ''),
			result      = CASE WHEN $5::text IS NOT NULL THEN $5::jsonb ELSE result END,
			error       = NULLIF($6, ''),
			completed_at = CASE WHEN $2 IN ('done','failed','cancelled')
			                    THEN NOW() ELSE completed_at END
		WHERE id = $1`
	if _, err := s.pool.Exec(ctx, q, id, status, sessionID, workDir, resultStr, errMsg); err != nil {
		return fmt.Errorf("update task %d: %w", id, err)
	}
	return nil
}

// FetchAgent calls rara-core GET /v1/agents/{id} and returns the daemon-relevant fields.
func (s *pgxStore) FetchAgent(ctx context.Context, agentID int) (AgentInfo, error) {
	url := fmt.Sprintf("%s/v1/agents/%d", s.coreURL, agentID)
	req, err := http.NewRequestWithContext(ctx, http.MethodGet, url, nil)
	if err != nil {
		return AgentInfo{}, err
	}
	req.Header.Set("Authorization", "Bearer "+s.coreToken)
	resp, err := s.client.Do(req)
	if err != nil {
		return AgentInfo{}, fmt.Errorf("fetch agent: %w", err)
	}
	defer resp.Body.Close()
	if resp.StatusCode != http.StatusOK {
		return AgentInfo{}, fmt.Errorf("fetch agent %d: status %d", agentID, resp.StatusCode)
	}
	body, _ := io.ReadAll(resp.Body)

	// Parse only the fields the daemon needs (bare-minimum: no full AgentRow import from rara-core).
	var row struct {
		ID           int             `json:"id"`
		Name         string          `json:"name"`
		Model        string          `json:"model"`
		MCPConfig    json.RawMessage `json:"mcp_config"`
		CustomEnv    json.RawMessage `json:"custom_env"`
		CustomArgs   json.RawMessage `json:"custom_args"`
		Instructions string          `json:"instructions"`
		Executor     string          `json:"executor"`
		SkillIDs     []int           `json:"skill_ids"`
	}
	if err := json.Unmarshal(body, &row); err != nil {
		return AgentInfo{}, fmt.Errorf("decode agent: %w", err)
	}

	var customEnv map[string]string
	if len(row.CustomEnv) > 0 {
		_ = json.Unmarshal(row.CustomEnv, &customEnv)
	}
	var customArgs []string
	if len(row.CustomArgs) > 0 {
		_ = json.Unmarshal(row.CustomArgs, &customArgs)
	}

	return AgentInfo{
		ID: row.ID, Name: row.Name, Model: row.Model,
		MCPConfig: row.MCPConfig, CustomEnv: customEnv, CustomArgs: customArgs,
		Instructions: row.Instructions, Executor: row.Executor, SkillIDs: row.SkillIDs,
	}, nil
}

// FetchSkills calls GET /v1/skills (list all) then filters + fetches files for each skill id.
// ponytail: full skill list then client-side filter — skills are few (~10s), not worth a new endpoint.
func (s *pgxStore) FetchSkills(ctx context.Context, skillIDs []int) ([]SkillBundle, error) {
	if len(skillIDs) == 0 {
		return nil, nil
	}
	need := make(map[int]bool, len(skillIDs))
	for _, id := range skillIDs {
		need[id] = true
	}

	// Fetch all skills
	url := s.coreURL + "/v1/skills"
	req, _ := http.NewRequestWithContext(ctx, http.MethodGet, url, nil)
	req.Header.Set("Authorization", "Bearer "+s.coreToken)
	resp, err := s.client.Do(req)
	if err != nil {
		return nil, fmt.Errorf("fetch skills: %w", err)
	}
	defer resp.Body.Close()
	body, _ := io.ReadAll(resp.Body)

	var rows []struct {
		ID      int    `json:"id"`
		Name    string `json:"name"`
		Content string `json:"content"`
		Trusted bool   `json:"trusted"`
	}
	if err := json.Unmarshal(body, &rows); err != nil {
		return nil, fmt.Errorf("decode skills: %w", err)
	}

	var out []SkillBundle
	for _, row := range rows {
		if !need[row.ID] {
			continue
		}
		bundle := SkillBundle{ID: row.ID, Name: row.Name, Content: row.Content, Trusted: row.Trusted}
		if row.Trusted {
			bundle.Files, _ = s.fetchSkillFiles(ctx, row.ID)
		}
		out = append(out, bundle)
	}
	return out, nil
}

func (s *pgxStore) fetchSkillFiles(ctx context.Context, skillID int) ([]SkillFile, error) {
	url := fmt.Sprintf("%s/v1/skills/%d/files", s.coreURL, skillID)
	req, _ := http.NewRequestWithContext(ctx, http.MethodGet, url, nil)
	req.Header.Set("Authorization", "Bearer "+s.coreToken)
	resp, err := s.client.Do(req)
	if err != nil {
		return nil, err
	}
	defer resp.Body.Close()
	body, _ := io.ReadAll(resp.Body)
	var rows []struct {
		Path    string `json:"path"`
		Content string `json:"content"`
	}
	if err := json.Unmarshal(body, &rows); err != nil {
		return nil, err
	}
	files := make([]SkillFile, 0, len(rows))
	for _, r := range rows {
		files = append(files, SkillFile{Path: r.Path, Content: r.Content})
	}
	return files, nil
}
```

```go
// CLIExecutor invokes the Claude Code CLI (`claude -p`) in the pre-built workdir.
type CLIExecutor struct {
	bin string // path to `claude` binary
}

// Run assembles the command, sets env/args from the agent config, spawns the CLI,
// captures stdout, and parses the JSON result. A non-zero exit is returned as an error.
func (e *CLIExecutor) Run(ctx context.Context, tc TaskCtx) (ExecResult, error) {
	// Build args: `claude -p "<instruction>" --output-format json [--model <model>] [custom_args...]`
	args := []string{"-p", tc.Instruction, "--output-format", "json"}
	if tc.Agent.Model != "" {
		args = append(args, "--model", tc.Agent.Model)
	}
	args = append(args, tc.Agent.CustomArgs...)

	cmd := exec.CommandContext(ctx, e.bin, args...) //nolint:gosec — bin is operator-controlled
	cmd.Dir = tc.WorkDir

	// Inherit the daemon's environment, then overlay the agent's custom env.
	cmd.Env = os.Environ()
	for k, v := range tc.Agent.CustomEnv {
		cmd.Env = append(cmd.Env, k+"="+v)
	}

	out, err := cmd.Output()
	if err != nil {
		return ExecResult{WorkDir: tc.WorkDir}, fmt.Errorf("claude exit: %w", err)
	}

	// Parse the JSON envelope claude returns in --output-format json mode.
	// Shape: {"session_id":"...","result":<any>}
	var envelope struct {
		SessionID string          `json:"session_id"`
		Result    json.RawMessage `json:"result"`
	}
	if jsonErr := json.Unmarshal(out, &envelope); jsonErr != nil {
		// Non-JSON output (e.g. plain text completion) — store as raw string.
		raw, _ := json.Marshal(string(out))
		return ExecResult{SessionID: "", WorkDir: tc.WorkDir, Output: raw}, nil
	}
	return ExecResult{
		SessionID: envelope.SessionID,
		WorkDir:   tc.WorkDir,
		Output:    envelope.Result,
	}, nil
}
```
